struct node 
{
    int data;
    struct node*left,*right;
};

int length()
{
    struct node *temp=front;int c=0;
    if(front)
    {
    while(temp!=rear)
    {
        c++;
        temp=temp->right;
    }
    c++;
    }
return c;
}
